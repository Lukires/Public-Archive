\begin{props}
    \prop{To podier}[]
    \prop{To musemarkører}[] Hver mus har en cursor
    \prop{To sigtekorn}[] Hver mus har et sigtekorn
    \prop{To knive}[] Hver mus har en kniv
    \prop{To ``Ready''-skilte}[]
\end{props}


\begin{sketch}
\scene{Der står to podier på scenen, et i hver side. Et sceneelement i hver side er spredt ud mod publikum så de peger mod publikum (lidt skævt så man kan gemme ting bag dem). På hvert sceneelement hænger der et ``Ready''-skilt. Bag hvert sceneelement ligger musemarkør, sigtekorn og kniv, og musene står parat.}

\scene{Ved start af hvert vers står N og G ved deres podier og rører ved deres mus (ikke danserne), så det er klart at de er gamere. I løbet af verset bevæger de sig væk fra podiet og er aktive på scenen på forskellige leder og kanter.}

\scene{Når en mus fragger en spiller, flasher der et rødt lys på scenen for at indikere en død.}

\scene{``Ready'' og ``Start'' er cirka det samme.}
\end{sketch}
