\begin{roles}
\role{3-4 studerende} der alle bliver dræbt af egernet.
\role{3-4 egern} \em{(en lille dukke, der kan sidde på en hånd)}
\end{roles}
\begin{props}
\prop{} Et par computere, borde, stole.
        De studerende skal desuden være udstyret med en kunstig hånd, så
        de kan styre egernet med den en hånd. Hånden skal være skjult under
        tøjet (helst en skjorte man kan åbne midt på maven). Det er meningen
        at egernet (der skal kun være et) skal dræbe alle studerende på
        scenen.
\end{props}

\begin{sketch}

\scene på scenen sidder nogle studerende og koder på en K-opgave

\says{TV-speaker} Vi afbryder den igangværende sketch, da det forlyder,
at det dødsensfarlige egern er set på Nørre Alle i København, nærmere
betegnet i nærheden af datalogisk institut ved Københavns Universitet. Det
forlyder, at egernet er bevæbnet med et nyt eksperimental-våben, som 
forsvaret netop har udviklet til udryddelsen af selv samme egern. Vi stiller
nu om til vores udsendte reporter på stedet.

\says{Reporter} Ja --- vi befinder os her på datalogisk institut. Egernet er
set herinde i selve bygningen. Der er dog ikke forlydender om nogen, der
er blevet maltrakteret, endnu\ldots Vent --- her er et par studerende. Hvordan
kan det være, at I sidder her og programmerer, mens et nødbringende ---
øeh --- dødbringende egern hærger instituttet?

\says{Stud. 1} Ja --- det er et temmeligt stort pres, men det er nu ganske 
simpelt: Vores opgave skal afleveres om 30 minutter. Man kan jo lejlighedsvis
tillade sig at citere H.C.Andersen "Gud giver os nødderne, men han knækker
dem ikke for os!"
\says{Reporter} Aha --- Men har I set egernet?
\says{Stud. 1} Jo, MM har set det. \em{(han peger på en anden studerende)}
\says{Reporter} Hvordan så det ud? Var det bevæbnet?
\says{Stud. 2} Jaeøh --- det var brunt og temmelig halerekursivt. Og ja - det
var skam bevæbnet. Desuden tror jeg, at jeg så en skindende metal-genstand som
den havde sidende i ryggen, jeg tror, det var en palet-kniv. Den var desuden 
udstyret med en anden palet-knivs-lignende genstand - jeg tror det var en
dejkskraber.
\says{Reporter} Aha - Hvad tror I grunden kan være til at egernet lige netop har
valgt Datalogisk Institut som tilholdssted?
\says{Stud. 3} [dumt] Øh --- kunne det være fordi, at den tror, her er en del
computere af mærket ACORN?
\says{Reporter} Det kunne da godt være\ldots Men hvad er dog det?

\scene{Egernet kommer til syne ved den bagerste af de tre studerende og går
helt amok \em{(ligsom en face-hugger fra aliens)}! Den studerende falder 
død om. \em{SQUEEK!} Egernet forsvinder lynhurtigt igen. Blod???}
\says{Reporter} [råber] Jamen, hvor er det dog forfærdeligt!

\scene{Egernet dukker frem hos den næste studerende, ordner ham og "zapper" 
over til den næste studerende og ordner ham.}
\says{Egern} Squeek! Squeeeek!
\says{Reporter} Hjælp!!! Hvor er det dog frygteligt!

\scene Lys dør hurtigt ud
\says{Egern} Squeek!
\says{Reporter} Shit --- den fik også lysmanden. Er der noget {\em
 nød}belysning?
\says{Egern} Squeek!
\says{Reporter} [Skrigende] Hjælp!! Hvor er {\em Nød}udgangen?? Argh!!!
\says{Egern} Squeek! Squeeeek!

\scene Slut!!

\end{sketch}
