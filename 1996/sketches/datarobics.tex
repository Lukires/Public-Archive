\begin{sketch}

\begin{roles}
  \role{1 datarobics instruktør} Iklædt alt muligt aerobics-outfit.
  \role{3 diku-nørder} Samme outfit, studiekort og nøgler.
  \role{1 fysiker} I blå skjorte og tykke Steen-briller.
\end{roles}

\begin{props}
  \role{Tre borde} Med en terminal på.
  \role{Cola-flasker} til nørderne.
  \role{Terminaler} Eventuelt bare et bord eller bare virtuelle :) 
\end{props}

\scene{På scenen er placeret tre nørder, der aktivt sidder og rykker frem i
  skuldrene højre højre venstre venstre armebøj armestræk. Colaflaske på
  bordet. Ind kommer instruktoren}

\says{Instruktør}[Rigtig overivrig og -gearet] Velkommen til datarobics. Vi
skal i dag sørge for at I alle får rørt jer lidt, det er jo ikke alle, der
har plads til at gøre gymnastik hjemme på kollegieværelset\ldots Vi
vil bede alle deltagere, også dem derhjemme, om at tage fat i musen. Ja kom nu
allesammen. Ja, sådan! (ad libitum) 

\says{Instruktør} Vi starter med tre lette museøvelser, først den lette: Og tryk og tryk og tryk og pause
\scene{Museøvelse 1, klikker på musen, først knap 1, derefter
  knap 2 og så knap 3}
\says{Instruktør} Og så skifter vi knap, den midterste: igen, ja kom!
\scene{Øvelse 1 igen}
\says{Instruktør} Og så den sidste knap, den til højre: igen, ja kom!
\scene{Øvelse 1 igen}

\says{Instruktør} Nu bliver det lidt sværere, for nu skal vi: click --- click-click --- drag
\scene{Museøvelse 2}

\says{Instruktør} Og så den svære, både den venstre og midterste knap: Og cut \ldots --- og paste
\scene{Museøvelse 3}

\says{Instruktør} Nu skal vi alle sammen lære at drikke cola, uden at få
ondt i skuldrene: Og drej --- og drik --- og skru --- og sæt
\scene{Colaøvelse 1}

\says{Instruktør} Så nu skal vi alle passe på, for nu kommer vagten: og vagt \ldots --- og væk igen!
\scene{Colaøvelse, man sætter colaen ned på gulvet og frem med studiekortet
  og vice versa}

\says{Instruktør} Click og vente \ldots og GIF
\scene{Folk lyser op, når billedet (!!!) kommer op på ``skærmen''}

\says{Instruktør} Return og CORE DUMP --- Return og CORE DUMP!!
\scene{Tastaturtryk og derefter tager man sig til hovedet}

\says{Instruktør} Og så lukker vi alle applikationerne ned på stedet.
\scene{Aktiv klikken hist og pist}

\says{Instruktør} Click vente vente vente vente vente vente MPEG
\scene{Et enkelt click og derefter tromme med fingerene på bordet, et
  latterligt udseende når natur-filmen kommer frem på skærmen}

\says{Instruktør}[Råber] Og HÅNDKØR!!

\scene Folk, undtagen fysikeren (begynder at håndkøre et program med pen og
papir) slipper computeren og begynder at knappe deres bukser op (hive
cykelshorts ned, imens fader lyset (hurtigt) ud)


\end{sketch}
