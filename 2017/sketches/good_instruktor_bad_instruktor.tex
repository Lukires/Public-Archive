\begin{roles}
\role{IG}[Ejnar] Instruktor (good)
\role{IB}[Bitre Mikkel] Instruktor (bad)
\role{F}[Simon] Person der bliver forhørt
\role{X}[Freja] Instruktør
\end{roles}

\begin{props}
\prop{Stol}[]
\prop{En kop kaffe}[]
\end{props}

\begin{sketch}
\scene{F sidder bagbundet på en stol.  IB kommer ind.}

\says{IB}[råber] Hvor er din ugeaflevering?

\says{F} Jeg har ikke lavet den!  Jeg sværger!

\says{IB} Det er din pligt at lave den!  Kom så med den!

\scene{IG kommer ind med en kop kaffe.}

\says{IG}[til IB] Jeg tager den herfra.

\says{IG}[til F] Vil du gerne have noget at drikke?

\says{F} Jotak, jeg vil gerne se vinkortet.

\says{IB} NÅ, DET ER BARE ÆRGERLIGT!

\says{IG} Nej, hør nu her, du kan roligt sige det til mig.  Jeg siger det ikke
videre.

\says{F} Jeg ved ikke hvad I snakker om.  Jeg lå bare og sov til forelæsning, og
lige pludselig kom I ind og bortførte mig hertil!

\says{IB} Vi har set dine tidligere ugeopgaver.  For satan i helvede, det kan du
gøre bedre!

\says{IG} Jeg er altså enig her... ``for helvede''.

\says{F} Jeg gør det bedre til eksamen, jeg lover det!

\says{IB} EKSAMEN ER I MORGEN!!!

\says{IG}[til IB] Måske har vi fået fat i den forkerte?

\says{IB} Men han står jo på Absalon!

\says{IB}[til F] Kan du i det mindste sige hvad kurset handler om???

\says{F} HVILKET KURSUS TALER I OM???

\says{IB} For helvede, du er jo kursusansvarlig!

\scene{Lys hurtigt ned.}
\end{sketch}
