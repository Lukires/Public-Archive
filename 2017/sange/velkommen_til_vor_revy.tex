\begin{song}
  \scene{Sceneshowet i denne sketch er meget inspireret af opførslen af ``Puttin' on the Ritz'' fra %
         Young Frankenstein. Se \url{http://youtu.be/w1FLZPFI3jc} for at få en idé om tanken bag det.}

  \scene{S prøver at sælge revyen til publikum. D får det til at falde til jorden.}

  \sings{S} Velkommen til vores show
            Hvor humoren den er så klog
            Her er en vits:
            \act{gestikulerer over mod D}

  \sings{D} \act{RÅBER!} DIKUREVY!

  \scene{S kigger skeptisk over på D, men fortsætter}

  \sings{S} Sketches med samfundskritik,
            (om) faglighed og politik
            Godkendt af Fritz

  \sings{D} \act{afbryder, RÅBER!} DIKUREVY!

  \scene{S forsøger at distrahere publikum fra D}

  \sings{S} Se blot vores fine rekvisitter!
            \act{Peger over på en meget uoverbevisende palme}
            Ingen tvivl om hvem det er der hitter \ldots
            \act{gestikulerer over mod D}

  \sings{D} \act{kaster glitter-konfetti ud over palmen} NU MED GLITTER!

  \scene{S ser opgivende ud og danser væk fra palmen}

  \sings{S} Sang og dans med intellekt
            (ry)tmisk sans og stort showmanship
            Til perfektion.
            \act{gestikulerer over mod D}

  \sings{D} \act{RÅBER!} DIKUREVY!

  \scene{Frustreret går S over og får D til at danse.}
  \scene{Akavet dans af en art}
  \scene{D ser en pige og går i panik, løber af scenen.}
  \scene{D danser hen til S, D og S danser en cool afslutning og bukker}

  \scene{KOMMENTAR FRA HYTTETUR: Ronni tager P med ud fra scenen og er classy}

  \scene{Lys ned, tæppe for.}
\end{song}
