\begin{props}
  \prop{4 barstole} ~
\end{props}

\says{Stemme} Vi skal nu se hvordan rigtige dataloger opfører sig ude
i byen, når de skal prøve at score rigtige piger på rigtige barer.

\scene{Der sidder to par i hver side af scenen på barstole}
\says{Datalog 1}Hej, skat, jeg hedder {\tt noerd@diku.dk}, du kan bare kalde mig Ib!
\says{Pige 1}Du læser datalogi, ikke?
\says{Datalog 1}Øh, ikke helt, jeg læser matematik \ldots som bifag. 
\act{Exit Datalog}

---
\says{Datalog 2}Hej skat, vil du med hjem og se mine .gif-billeder?
\says{Pige 2}Du læser datalogi, ikke?
\says{Datalog 2}Hvad med .pony? .ged?
\act{Exit pige}

---
\says{Datalog 1}\act{Lægger an til en pick-up line}
\says{Pige 1}Du læser datalogi, ikke?
\says{Datalog 1}Nå, men jeg tror også, at der er åbnet for ftp nu\ldots
\act{Exit datalog}

---
\says{Datalog 2}(Myndigt) Hej skat, jeg er læge. Der ser ud til at være noget i vejen
med dit underliv. Ska' jeg ikke lige chekke?
\says{Pige 2}(Naivt) Er du gynækolog?
\says{Datalog 2}(Påtaget uskyldig) Jah, da!
\says{Pige 2}(Mistænksomt) Hmmm. Du læser datalogi, ikke?
\says{Datalog 2}Jeg tror nok at cola-automaten er blevet fyldt op igen\ldots
\act{Exit Datalog}

---
\says{Datalog 1}Hej Babe, hvilket styresystem er du til?
\says{Pige 1}Du læser datalogi, ikke?
\says{Datalog 1}Nej, jeg læser inf.psyk.
\says{Pige 1}Så LÆSER du altså datalogi!
\says{Datalog 1}Nej, minsandten! Jeg kan høre en maskine, der er ved at
gå ned\ldots
\act{Exit datalog}

---

\act{Datalog 2 sidder og ser trist ud og kigger væk fra Pige 2}
\says{Pige 2}(Sensuel) Hej skat, gi'r du en drink?
\says{Datalog 2}(Oplivet) \act{Vender sig om mod Pige 2}
 Ja, selvfølgelig! Hvad vil du ha'? ROM, Blå Nykøbing\ldots ?
\says{Pige 2}Du læser datalogi, ikke?
\says{Datalog 2}JO, OG JEG ER Stolt Af det!
\act{Exit Pige}

---
\says{Pige 1}Hej! Jeg er humanist. Jeg kan en helt masse. Jeg kan græsk,
spansk, svensk, dansk og fransk. \act{Smiler, så man kan se både et par sensuelle røde læber og et Colgate(tm)-tandsæt}
\says{Datalog 1}Jeg kan også en helt masse. Jeg kan Emerald, Standard-ML
og LISP. Skal vi mødes ovre på Parentesen?
\says{Pige 1} Du læser datalogi, ikke?
\says{Datalog 2} \act{Kigger væk, opgivende} Jo.

---
\says{Datalog 2}Hør, jeg kan lige så godt sige det fra starten. Jeg læser datalogi.
\says{Pige 2}Fedt nok! Så kan du måske fortælle mig, hvorfor min printer ikke vil
skrive danske karakterer?
\says{Datalog 2}\act{Kigger frem} (Undrende) Karak\ldots \act{Kigger igen mod Pige 2} (Undvigende) Det er jo lidt maskinafhængigt, sådan noget med danske 
TEGN\ldots
\says{Pige 2} \act{Begynder at tage vildt på den stakkels, uskyldige datanørd} Jamen, så tjener du da i hvert fald en masse penge, når du er færdig!
\says{Datalog 2}(Skeptisk) Du læser fysik, ikke?
\says{Pige 2}\act{Bliver endnu vildere} (Siger glad) Ja, hvordan vidste du det?
\scene{Datalog 2 vrider sig løs og flygter}
