\begin{sketch}
  
  \scene Efter sangen fortsætter forhøret

  \says{FH}[Lægger hånden på Everts skulder og sætter ham ned] Jaja, det er
  meget godt, men det er stadig ikke pænt gjort!
  
  \says{GS} Jamen Fritz, han gør jo også lidt for de studerende. Han
  lader dem bruge auditoriet, ja og så er der kommet fine nye
  brugervenlige farver på opslagstavlerne
  
  \says{FH}[råber i hovedet på EV] Ja, er du homo?!
  
  \says{EV}Jeg tror godt jeg vil ringe til min advokat.

  \says{GS}[hjælpsom. peger] Ja, telefonen er lige derude

  \scene{EV begynder at rejse sig}

  \says{FH}[spydigt] Jeg håber du har et Danmøntkort.

  \says{IS}[råber fra salen i megafon] Halløj! I \emph{har} brugt den tid sketchen
  skal tage!

  \says{FH}[falder ud af rollen] Nå? Ja, så må vi gøre det lidt hurtigt

  \scene{Replikkerne skal komme relativt hurtigt herfra. Gerne
  dukketeateragtigt og med meget markerede udtaler}

  \says{FH} Nå, hvor er Stig?

  \says{EV} Det siger jeg aldrig!

  \says{FH} Årh, kom nu. 

  \says{EV} Ok, han er i kælderen.

  \scene{ST kommer ind af sig selv og ser sur ud}

  \says{GS} Hej Stig. Godt vi fandt dig. Har du det godt?

  \says{ST}[sur] Jeg har ligget bundet i kælderen i en uge.

  \says{GS}[overhører det] Nå, det var godt.

  \says{FH} Evert, hvorfor gjorde du det?

  \says{EV} Jeg havde en traumatisk barndom.

  \says{FH} Nå, skam dig. Lad være med at gøre det igen.

  \says{EV} Det skal jeg nok.

  \says{GS} Fandt egentlig vi nogensinde Jyrki?

  \scene{Tæppe for hurtigt. Efter en kort pause}

  \says{JY}[pr. voice over] Hallooooooo? Hemupgifter!

\end{sketch}
