  \prop{Afslappet tøj} FA har kasket på, rusvejleder-T-shirt, shorts

  \prop{Attach\'emappe} Til MB
  \prop{Lejetøjsguitar} Til FA

\scene{Bandet begynder at spille forspil til ``Kender Du Det''. FA
  kommer på scenen. Han er lystig og traver halv-dansende rundt\ldots{}

\act{FA danser rundt på scenen, hilser på folk
  blandt publikum etc. mens han synger.}

\scene{  Indtil\dots}
}

  \sings{FA}[omkvæd]

\scene{Nu kommer MB ind. Han har rank ryg og næsen lidt i
  sky. Overlegen og ``hånd-på-skulderen-det-skal-nok-gå''-agtigt}

\scene{FA er i mellemtiden trådt endnu et skridt tilbage, står midtfor.}

\scene{SLI slæber sig skrådt tilbage til venstre\ldots{} giver plads
  til MB, der stiller sig fremme, en smule til højre på scenen.}

\begin{song}[Ode an die Freude]
  \sings{MB}
  Det var første dag på DIKU
  og jeg skulle ha' DAT-0.
  Jeg har masser tid til fester
  men Caf\'een?$^{TM}$ er et hul.

  \sings{MB}[omkvæd]
  Der er intet jeg har svært ved.

  Der er folk der kæmper hårdt, men \act{kigger evt. hen på SLI}

\scene{Sang stopper og FA træder ind imellem de to andre. Lægger
  hånden på skulderen af SLI, trækker ham frem på scenen efter sig,
  lægger den anden hånd på MBs højre skulder.}

\scene{Alle tre burde nu stå fremme og midtfor på scenen.}

\begin{sketch}
  \says{FA} Heeeej - var vi ikke på rustur sammen? Det er da længe
  siden\ldots

  \says{MB}[tørt] Ja, det vil så være 2 år og 2 måneder siden
  \act{trækker sig lidt væk -- væk fra berørelsen}.

  \says{SLI}[opgivende] \act{tager sig tl hovedet, og ryster derved
  FAs hånd af} Er der allerede gået 2 år?

  \says{FA} Ja ja \dots tiden flyver når man har det sjovt. Jeg er
  rusvejleder og skal være med i Revyen$^{TM}$.\\
  Skal vi ikke gå på Caf\'een?$^{TM}$?
  
  \says{SLI} Eeh\dots jeg har desværre ikke tid. Jeg er i gang med at
  skrive oversætter.

  \says{MB} Oversætter? \act{tænker}\\
  Nåeh\dots videregående oversætterteknik.

  \says{SLI} Nej, jeg er igang med oversætteren for 2. gang. Jeg håber
  jeg består både 1E og 1F her til vinter.

  \says{FA} \act{Klasker til SLI} Aaaah, det er ligesom mig\ldots det
  vil sige - \act{bob-bob}\\
  \ldots{} nu skal du høre:
\end{sketch}

\scene{MB træder frem og lidt væk fra de andre to. SLI og FA bør stå

\begin{song}[Ode an die Freude]
  \sings{MB}

  men jeg er jo også hård.

\scene{MB træder til siden, hen til de andre to (FA og SLI), der
  samtidig er trådt ligeså langt frem som MB.}

\begin{sketch}
  \says{MB} \ldots{} og sådan blev jeg kandidat. \act{SLI overrasket
  og lidt modløs.}

  \says{MB} Nu skal jeg så bare \act{Bliver afbrudt af FA} finde ud af, hvad
  jeg så \ldots
  
  \says{FA}[Begejstret] Gjorde du det? TILLYKKE! Ja jeg har lige fået
  dispensation\ldots{} \act{overkæk talkshow-host bevægelser} IGEN!
  \act{griner}.

  \says{SLI}[Lidt glad] Jeg er lige blevet optaget på mit bifag
  \act{lidt stolt}.
  
  \says{FA} Jamen læste du ikke \act{begynder at smådanse} det der
  \ldots{} \act{knipser med fingrene, danser}

  \says{MB} Dansens Æstetik og Historie.

  \says{SLI}[undskyldende] Jooo, men det droppede jeg.

  \says{FA} Nå ja! men så læste du jo\ldots{} \act{holder fingrene op
    og danner ``vindue''.}

  \says{MB}[lidt utålmodigt] Film \& Medievidenskab!

  \says{SLI}[undskyldende] Jo - det er også rigtigt, men nu er jeg
  altså blevet optaget på Informationspsykologi \act{lyser op, glad}.

  \says{FA}[tænkende] \act{træder lidt væk fra de andre -- spekulerer
  ud til publikum} Hmmm\ldots{} det ku' da være man sku' prøve det der
  Fuldtids Datalogi. \act{går et skridt frem på scenen}
\end{sketch}
