\begin{sketch}
\scene{Når publikum kaldes ind fra pausen, spilles der temaet for
  \texttt{M*A*S*H}.  På projektor vises teksten \texttt{B*A*S*H} i
  \texttt{M*A*S*H}-skrifttype}

\scene{O, L og A står på scenen og drikker hjemmebrygget øl.}

\scene{Der flyver H ind fra midterpublikumindgangen på bandsiden, og flyver(ved at dreje rundt om sig selv med spredte arme og lave helikopterlyde) ned igennem mellemgangen og ned til scenen}

\scene{H kommer rullende ind med en datamat der ryger.  Alting bliver pludselig
  \textbf{meget} dramatisk.  Musikken stopper.}

\scene{S kommer ind.  Under sketchen taber hun gentagne gange instrumenter og er
  nødt til at bukke sig ned.  Måske lægger O lidt an på S.}

\scene{O, L og A går hen til H.}

\says{H} Netkortet er kørt over af en bus!

\says{O} Tjep!  På bordet!

\scene{L og A tager hurtigt men roligt datamaten op og lægger den på et
  operationsbord altimens S giver O handsker på.}

\says{O}[lytter til datamaten med et stetoskop] Blæseren er røget!

\scene{O puster gentagne gange ind i datamaten, der hvor blæseren sidder.}

\says{O}[til S] Giv den nitrogen hver halve minut. 
\says{O}[til L] Tilslut netkablet. 
\says{O}[til A] Giv den nød-strømforsyningen og bedøv den med Ctrl+Z.

\scene{L tilslutter et netkabel. På projektor ser man pings.  Man hører pings
  med længere og længere distance.}

\begin{verbatim}
PING datamat.local (127.0.0.3) 56(84) bytes of data.
64 bytes from datamat.local (127.0.0.3): icmp_req=1 ttl=53 time=4.33 ms
64 bytes from datamat.local (127.0.0.3): icmp_req=2 ttl=53 time=10.38 ms
64 bytes from datamat.local (127.0.0.3): icmp_req=3 ttl=53 time=100.36 ms
...
\end{verbatim}

\says{O} Den flatliner!  Giv den kølerchok!

\scene{S hælder flydende nitrogen ud over den.}

\scene{O vifter nitrogenen væk.}

\says{O} Skruetrækker.

\scene{S går hen for at hente en skruetrækker. Kigger forvirret på en væg med mange
       skruetrækkere, alle markeret med stjerne-skruetrækker-symbolet bortset fra
       én.}

\says{O} Hvornår kommer den skruetrækker???

\scene{S tager den ene ikke-stjerneskruetrækker og giver den til O.}

\says{O} NEJ!  Stjerneskruetrækkeren!

\scene{A skubber S væk og giver glad O en stjerneskruetrækker.}

\scene{O skruer hurtigt men forsigtigt bagpladen af datamaten.  Han smider
  bagpladen væk.  Han holder sin hånd over bagsiden.  Man hører blæseren pumpe
  derudaf.}

\says{O} Bedøv datamaten! Den har brug for en... ram-transplantation.

\scene{A løber over for at ringe efter RAM. S checker kortvarigt og kommer tilbage.}

\says{S} Vi er løbet tør for bedøvelsesmidler!

\says{O}[seriøs mine] Vi hotswapper rammen.

\says{L} Men det kan man da ikke!

\scene{O griber fat i Ls krave.}

\says{O}[til L] Jeg har ikke studeret datalogi i 12 år for at få det at vide!

\scene{O begynder at tage rammen ud.}

\scene{SC ruller en datamat ind.}

\says{SC}[peger på den nye datamat] Jaeh, den havde et clockcykeluheld på
informationsmotorvejen.

\scene{S giver O den nye datamat. L skubber Thinkpad'en til siden.}

\scene{O begynder at arbejde på den nye datamat. A sætter kabler til den. S tørrer sveden af Os pande. L arbejder på ThinkPad'en.}

\says{L} Den får feber igen!

\says{O} \act{Hopper tilbage til Thinkpad'en} Kølerchok!

\says{S} Vi har ikke mere flydende nitrogen!

\says{O} Blæs!

\scene{A tager blæsebælger og puster, L, S og O puster med munden.}

\scene{Der stopper med at komme pings.  Projektor: \texttt{Packet dropped.}}

\says{S}[til O] Den taber pakker!

\scene{O hopper op på bordet og prøver en sidste omgang chipmassage.}

\says{O}[råber] GENSTART! GENSTART!

\scene{O kravler stille ned og ryster på hovedet.  De andre er kede af det.}

\scene{O tager sine briller af.}

\says{O} Dødstidspunkt: 13:37.

\says{A}[casual] ... Så må vi købe en ny.

\scene{Alle går ud. Datamaten bliver liggende på operationsbordet til næste nummer.}

\end{sketch}
