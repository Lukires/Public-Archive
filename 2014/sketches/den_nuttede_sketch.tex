\begin{sketch}

\scene{Denne sketch er båret frem af fysisk humor. Når R siger noget, tøffer R rundt imens. I'erne er så vidt muligt statiske. 3 prikker indikerer at det angivne talemønster looper. I'erne peger på R når til taler til ham.}

\scene{Fra sceneindgangen og mod bandscenen står I'erne på kasser i følgende rækkefølge: I3, I2, I1, I0}

\scene{R tøffer ind på scenen forbi I'erne.}
\says{R}[Begejstret] Java, Java, Java\ldots

\scene{Når R når til I0:}
\says{I0} SML!
\says{R}[væmmes] SM?!
\says{I0} SML!
\says{R} SML, SML, SML\ldots

\scene{R løber i cirkel bag om I'erne og når R når frem foran I1:}

\says{I1} C!
\says{R} C, C, C\ldots

\scene{I0's blik følger R hele vejen rundt imens R løber i cirkel bag om I'erne og når R når frem foran I3:}

\says{I3} Assembler!
\says{R} Ad! \act{løber videre} C, C, C\ldots

\scene{I0 begynder at trippe med foden når R forlader I3. Når R når til I0:}

\says{I0} SML!
\says{R}[ubegejstret] SML, SML, SML\ldots

\scene{R løber igen bag om I'erne, og når han når mellem I1 og I2 kigger han frem mellem dem}

\says{R}[begejstret] C, C, C\ldots

\scene{R løber igen videre, og når han når til I2:}

\says{I2} Haskell!
\says{R} Haskell, Haskell\ldots \act{løber videre et par trin}
\says{I2} Prolog!
\says{R} \act{Går tilbage foran I2} Prolog?
\says{I2} Haskell, Prolog, Erlang!
\says{R} \act{Løber videre} Haskell, Erlang \act{snubler og falder helt}, Prolog?
\says{R} \act{Rejser sig og løber videre i stigende tempo} Haskell, Prolog, Erlang, Haskell, Prolog, Erlang\ldots

\scene{R løber to omgange, og når han kommer ind foran igen:}

\says{R} SML, C, Haskell, Erlang, Prolog! \act{rejser armeme begejstret}

\scene{B kommer ind på scenen og giver R hånden}

\says{B} Java!
\says{R}[trist] Java, Java, Java\ldots \act{forlader scenen}

\scene{Lys ned.}
\end{sketch}
