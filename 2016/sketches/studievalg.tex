\begin{roles}
    \role{G}[Simon] Gymnasieelev; har \emph{meeeeget} svært ved at vælge studieretning.
    \role{S}[Sebbe] Studievejleder Datalog/Kemiker/Biolog/Tør fysiker; står ved datalogi-bordet ved åbent hus-arrangement; skifter mellem alle fire typer
    \role{X}[Brainfuck] Instruktør
\end{roles}

\begin{props}
    \prop{Plakat med teksten "Danmark har brug for \{datalogi, kemikere, biologer\}", billede af en pæn studerende og citatet ``Det er spændende!''}[]
    \prop{Plakat med teksten "Danmark har brug for fysikere", billede af en tør studerende og citatet ``Jeg er gymnasielærer!''}[]
\end{props}


\begin{sketch}

\scene{Efter hver ding-lyd skifter S kostume mellem de forskellige studieretninger.}

\scene{G går frem og tilbage på scenen}

\says{G} Æh hvor er - det - bare - \emph{svært!} at vælge hvad jeg skal studere efter gymnasiet!

\scene{AV afspiller en ding lyd.  På OverTeX er en plakat for datalogi:  ``Det er spændende!''}

\says{G} Nøjjjjj! Datalogi! Det lyder spændende! Se selv, det står på plakaten! Åhr! Og der er åbent hus! Det skal jeg ind til!

\scene{Scenen bliver til åbent hus, G går over til datalogi. S står ved bordet.}

\says{S} Hej! Er du interesseret i at høre lidt om datalogi?

\says{G} Ja! Jeg har hørt at det skulle være spændende! Er det rigtigt?

\says{S} Ja, det er \emph{vildt} spændende! På datalogi får du lov til at lære om \ldots

\says{G} Nøøøøøøj! Så det er ikke bare spændende, men ligefrem \emph{vildt} spændende?!

\says{S} Ja, vi lærer for eksempel \ldots

\says{G}[går væk fra S] Nøøøøøj! Jeg skal så meget læse datalogi efter gymnasiet! Det lyder så nice!

\scene{AV afspiller ny ding lyd.  Plakat med Biologi.}

\says{G} Nøøøøøøj! Er biologi \emph{også} \emph{spændende}!? \act{Går hen til S.} Jeg troede at det var datalogi der var spændende?

\says{S} Ja, biologi er vildt spændende! Har du nogensinde overvejet \ldots

\says{G}[går væl fra S] Nøøøøøøøøøj! Gad vide om jeg kan læse både datalogi og biologi?

\scene{AV afspiller ny ding lyd.  Plakat med Kemi.}

\says{G}[falder ned og spraller på gulvet] Nøøøøøøj! Ej, jeg gad vide hvor mange fag jeg kan tage? Jeg har bare lyst til at tage det heeeele!

\scene{G rejser sig op og går mod S.  Mens han går afspiller AV en ding lyd og en plakat med fysik kommer op med teksten:  ``Du kan blive gymnasielærer!''}

\says{G} Arh, okay, der er måske grænser...

\scene{Lys ned.}

\end{sketch}
