\begin{props}
\prop{FARTIS}[] TARDIS/Portapotty
\end{props}

\begin{sketch}

\scene{Mørk scene.  Dunkel musik.  Lyden af et toilet der skyller ud.  Lys op
og TARDIS/portapotty (FARTIS) på scenen, med godt med røg på scenen.
Konstante pruttelyde i baggrunden.}

\scene{DP træder ud af FARTISen.}

\says{DP}[vifter sig for næsen og trækker sine bukser helt op] Puh!  De der
skaldyrspizzaer fra Duniyas...

\scene{DP opdager publikum.}

\says{DP} Hov!  \act{pause, kigger sig omkring} Det er jo store UP-1!  Jeg
er Dr. Prut, og med denne \act{vinker mod FARTISen} rejser jeg gennem tid og
rum i jagten på min ærkefjende, Rektor Røv!

\says{DP} Ganske vist rejser jeg rumligt mest i Store UP1, og
tidsmæssigt mest fremad, men jeg har en kløe et vist sted der siger mig,
at jeg meget snart finder-

\scene{RR træder ind!}

\says{RR} Rektor Røv!  Det er mig, der er Rektor Røv!

\says{DP} Rektor Røv!  Jeg har en høne at plukke med dig.

\says{RR} Dr. Prut!  Øjeblik!  Jeg skal lige slå en prut!

\scene{Rektor Røv slår en prut!}

\says{DP} Er du færdig?

\says{RR} Nej, jeg skal lige slå en til!

\scene{RR slår DP.}

\scene{DP falder.}

\says{DP} Ha!  Den her afbødede mit fald!  \act{Dr. Prut tager en pruttepude
frem.}  Uden den var jeg virkelig faldet på røven.

\says{RR} Du var næppe faldet på mig, min fjende, Dr. Prut!  Og nu
stjæler jeg din tidsmaskine, så du aldrig kan fange mig!

\scene{RR stormer ind i FARTISen.}

\says{RR} Hvilken teknologi!  Den er jo endnu mindre indeni end den
ser ud udenpå!  Nu rejser jeg frem i tiden!

\scene{Pruttelyd.  Der går fem sekunder.  Mere røg, og lyden af et toilet der
skyller ud.}

\says{RR} Ha! -Vent, hvad?!

\says{DP} Du rejste kun fem sekunder frem i tiden!  Nu skal du få røven
på komedie, din skiderik!

\scene{DP hopper på RR, de slås og ryger ind i maskinen, hvor
døren lukkes.  En høj pruttelyd lydes!}

\says{DP} Dit fjols!  Du har lavet et stort skyl!  Nu sender maskinen os
helt ind i den fjerne fremtid!

\says{RR} Til hvornår?

\says{DP} Helt ind i næste nummer!

\scene{Lys ned.}

\scene{Lys op.}

\scene{Pruttelyd, røg, og toiletskyl.  Døren til FARTIS åbner.  Sangerne fra næste nummer står forvirret på scenen.}

\says{RR} Hvor... er vi endt?

\says{DP} Det handler ikke om hvor... men hvornår.  Vi kan aldrig vende
tilbage til vores egen sketch.  Og vi får næppe lov at ødelægge endnu et nummer.

\says{RR} Lort!

%\scene{Lys ned.}

\end{sketch}
