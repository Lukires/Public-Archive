\role{X}[Bette-Mikkel] Instruktør

\begin{props}
\prop{Kam -- bonus hvis det er en springkam}[]
\prop{Papkasse med DANGER-klistermærker}[]
\prop{En gammel router}[]
\prop{En kasse med påskriften ``TO DO''}[]
\prop{Et bord}[]
\prop{Et kaffekrus}[]

\prop{A/V 0: Gammeldags telefon som ringer}
\prop{A/V 1: Det banker på døren}
\prop{A/V 2: En elektrisk zap-lyd}

\end{props}


\begin{sketch}

\scene{Dramatisk skift til SCIENCE ITs kontor.}

\scene{\textbf{Akt 2}: Ordren.  Dette foregår i den anden side af scenen, mens
den første side er mørklagt.}

\scene{G sidder og drikker kaffe.  Telefonen ringer[A/V 1], og G tager den.}

\says{G} Javel, javel!  Det skal vi sørge for!  Godt, farvel.  Farvel!
\act{lægger røret på}

\says{G} U!  Kom herind -- vi skal lave cyberforsvar.

\scene{U kommer ind og kæmmer sit hår på vejen.}

\says{U} Ja, ja.  Ja vi skal!  Skal vi installere Service Pack hos Dronningen?

\says{G} Nej nej, det er lidt ligesom, hvis du skulle installere Service Pack
-- for hele Danmark!

\says{U} Gisp!  Jamen så er det jo vigtigt! \act{pakker straks sin kam væk}

\scene{Det banker på [A/V 2].  De henter en pakke.}

\scene{G pakker æsken op.  Den indeholder en gammel router/switch i en dristig
rød farve og med glimmer.  Der er også bobleplast i.}

\says{G} AAAH, dejligt!  Cyberforsvarsdimsen!  Den kan køre både Norton
Antivirus og McAfee, på samme tid!

\says{U} Hvad er verdenen ikke kommet til.

\scene{St kommer ind med en Mac hvor skærmen er separeret fra tastaturet.}

\says{St} Jeg kan ikke logge ind på Facebook med min Mac.

\says{G} Det fikser vi.  U, hold lige SKARPT øje med CFD'en!

\scene{U stirrer kraftigt på CFD'en.}

\scene{G tager Mac-resterne og studerer dem nøje og prøver at sætte dem sammen.
G tager en kontaktlim frem og skal til at håndværke.}

\scene{St forlader scenen.}

\scene{G kigger for at se om St er helt væk.  G tager så en papkasse med et
TODO-skilt frem og smider Mac'en derned.}

\says{G} Pyhae, nu skal jeg have kaffe.

\scene{G går ud.}

\scene{U kan ikke dy sig for at pille ved CFD'en.  U forbinder et kabel
fra den ene ende af CFD'en til den anden og brænder derved maskinen af [A/V 2], så der
kommer hvid røg op af den.}

\says{U} Gisp!

\scene{Der kommer fortsat røg op.  U forsøger at fange røgen ind og putte den
tilbage i maskinen. Når U forsøger at standse ``blødningen'' med sin hånd, så
kommer der bare røg ud et andet sted.}

\scene{U smider et hvidt klæde hen over CFD'en for at skjule den.}

\scene{G kommer ind igen.}

\says{G} Lad mig lige kigge engang --

\says{U}[nervøs] Hey, har du set at man kan prøve Windows 10, helt gratis...
derhenne! \act{peger på TeXnikken}

\says{G} Jooeh, det vil jeg, \textbf{efter} vi har konfigureret CFD'en!

\says{G}[smilende] Jeg kommer lige hen og kigger på den.

\says{U} Nej, bliv derovre!  Du bryder min... øh... intimsfære!

\says{G} Huhuhuh, nu lyder du som min hustru.

\scene{G tager en tår af sin kaffe, men brænder sig.}

\says{G} Uha, varm! \act{lægger kaffen ned i TODO-kassen.}

\says{G} Nej, nu VIL jeg altså gerne se den!

\says{U} Det KAN du ikke!  Jeg er ved at stress-teste den, så øh... du
skal lade den være!  Den er ret stresset, forstååås...

\scene{U kigger skamfuldt ned i jorden, med hænderne på ryggen, og med den ene
fod nervøst drejende i gulvet.  U prøver at sige noget, men kan ikke få sig selv
til det.  U får taget sig sammen.}

\says{U} Jeg har noget jeg gerne vil sige.

\scene{G står og fedter med sin kaffe.}

\says{U} Ser du, CFD'en er vist blevet en smule... shell-chokeret.

\scene{G standser stift i sit arbejde.}

\says{G} Hvad er det du prøver på at sige, U?

\says{U} Jeg har vist ødelagt den... lidt?

\scene{G taber det værktøj han har i hænderne.}

\says{G} HVAD HULEN, HAR DU ØDELAGT DEN?!!?!!?

\scene{U skælver af skræk for G's vrede.}

\says{U} Det var ik' med vilje, det var ik' med vilje!

\says{G} JEG VIL SIMPELTHEN IKKE STÅ MODEL TIL DENNE DANMARKS-TERROR!  DER
\textbf{MÅ} VÆRE EN STRAF!

\says{U} Nåde, nåde og be' be'.

\scene{U falder på knæ.}

\says{G} DU SKAL...

\says{G} ... HENTE KAFFE TIL MIG, PROMTE OG FLUKS!

\scene{U hopper muntert op fra gulvet.}

\says{U} Ih ja, det skal jeg nok!  Med det gode sukker?

\says{G}[syngende] Jaaaahr.

\says{U} ... og fløde??

\says{G}[syngende højere] Jaaaeehr!

\says{U} Javel! \act{gør honnør og marcherer muntert ud af scenen}

\scene{Kort pause.  G kigger på CFD'en.}

\scene{G flytter muntert CFD'en ned i TODO-kassen og fløjter.  G klapper i
hænderne og forlader scenen.}

\end{sketch}
