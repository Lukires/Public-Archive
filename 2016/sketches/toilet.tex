\begin{sketch}

\scene{Lys op.  Scenen er tom.  T løber ind.  T har sin datafon klar og taster et nummer.}

\says{T}[i telefonen; tissetrængende] Jeg vil gerne bestille et toilet!

\says{P}[uden for scenen] Velkommen til Københavns autoriserede toilettjeneste.  Du taler med Preben.

\says{T}[i telefonen] Preben, jeg skal tiiisssseee!  Kom til Universitetsparken 1!!!

\says{P}[uden for scenen] Er der om ti minutter!

\scene{T hopper tissetrængende rundt.  T kigger på sit ur, og man forstår at de ti minutter er gået.  P kommer langsomt ind med et toilet.}

\says{T} Endeligt!

\scene{T skal til at bruge toilettet.}

\says{P} Øjeblik!  Jeg skal lige starte tissemeteret.

\scene{P skruer langsomt på et tissemeter.}

\says{P} Så er det klart.

\scene{T går på toilettet og tisser i noget tid.  Efter lidt tid kommer der lyden af et toiletskyl.}

\scene{T kommer ud helt afslappet.}

\says{T} Det var dejligt!

\says{P} Det bliver \act{kigger på tissemeteret} 201 kr.

\says{T}[betaler] Det var sørme dyrt!

\says{P} Jo, måske, men-- \act{afbrydes}

\scene{J løber ind.  J har solbriller og åben skjorte.}

\says{J} JA, det vil jeg godt nok også sige!  DYRT, DYRT, DYRT!

\scene{J skubber P og P's toilet ud af scenen.}

\says{J}[mod publikum, prøver at få dem med]
DYYYYYYYYYYYYYYYYYYYYYYYYRT!!!

\says{J} BRUG DOG MIN APP I STEDET!

\says{J} Visitkort!

\scene{J kaster med sig med visitkort og løber så ud igen.  Visitkortet kommer også på OverTeX.}

\says{T}[tager et kort op og kigger på det] Men nu har jeg jo tisset.  Æv...

\scene{T tager sig pludselig til maven.}

\says{T}[glad] Men så er det jo godt jeg også skal \emph{skide}!

\scene{T tager smartphone frem og skal til at ringe.  T tager sin tommelfinger frem og trykker ukritisk på midten af smartphonen.}

\scene{J kommer med det samme løbende med et toilet.}

\says{J} SÅ ER DER TOILET!

\says{T} Det var sørme hurtigt.

\scene{T venter akavet.}

\says{J} Gå dog ind og skid!

\says{T} Men skal du ikke starte tissemeteret først?

\says{J}[vinker T på vej] Nejnej, det er fiiiiiint!

\scene{T går ind og skider.  J tager sin smartphone frem og scroller lettere manisk op og ned på den.}

\says{T}[fra toilettet] Hvordan kan du egentlig være så hurtig?

\says{J} Det er deleøkonomi!  Intet sløvende monopol!  Alle pengene går til mig direkte!  I hvert fald de fleste--  HEY, har du set hvor pæn vores app er?  Den SHINER!

\says{T} Ja... det er også et meget pænt toilet du har.

\says{J} Tak du!  Det er jo mit eget.  Deleøkonomi, du ved!  Jeg kan jo ikke sidde derhjemme og skide hele dagen!

\says{J} Så husk at gå ind på appen og anmelde mit toilet.

\scene{Lyden af toiletskyl afspilles.  T kommer straks ud af toilettet.}

\says{T} Det var dog dejligt.

\says{J} Det bliver \act{trykker med tommelfingeren på sin smartphone} 77 kr.!

\says{T} Sikke billigt!

\says{J} Jepsi!  Nu må jeg afsted igen, nogen på Amager har voldsom
diarré!

\scene{J sprinter ud af scenen med sit toilet.}

\says{T} Ham Johnny har da fart på!  Ikke som ham den slatne Preben og hans kedelige toilet!

\scene{T tager en dyb indånding.  Scenelyset dæmpes, og der kommer spot på T.}

\says{T} Jeg er overbevist.  Deleøkonomi og brugervenlige apps er fremtiden!  Ukreative standard-toiletter og dovne, monopolistiske chauffører med fjollede privilegier er fortiden!  Aldrig mere-- \act{afbrydes}

\scene{T får en besked.}

\scene{AV: Johnny fængslet.}

\scene{Lys ned.}

\end{sketch}
