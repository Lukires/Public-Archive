\begin{sketch}

\scene{I denne sketch kører vi 4 parallelle sketches; alle 4 ens. Hver sketch
har en A, B og C, og disse sketches kører synkront, med ens bevægelser på
scenen, bare på 4 forskellige steder. Scenen skal gerne se fyldt ud når skuespillerne står på den.}

\scene{Scenen starter tom. I kommer ind.  I vinker A* og B* ind.}

\says{I} Kom bare!

\scene{Alle A* og B* kommer ind og stiller sig klar til deres sketches.}

\says{I} Kør!

\scene{I står i hjørnet og holder øje.}

\scene{Hver C går sexet forbi det tilhørende A-B-par.}

\says{A} Har du set hende der?

\says{B} Åhhhrr ja, hendes nederdel er *godt* *nok* hul\emph{kort}.

\scene{R løber ind.}

\says{R} Stopstopstop, nejnej, \act{til I} jeg sagde at jeg ville have \emph{EN GO' SKETCH}!

\says{I} Nååååå! \act{slår sig til panden} Jeg troede at du ville have en ENGROS SKETCH!

\scene{LYS NED.}

\end{sketch}
