\begin{props}

\prop{Gummihandsker}[]
\prop{Stetoskop}[]
\prop{2 komplet ens business-suits}[] Snak med Mark og Valiant.
\prop{2x Heksedoktor-maske/-fjerhat}[]
\prop{4-6 Shamanistiske halskæder}[]

\end{props}

\scene{Tæppe fra, lys op}
\act{SO} Står og er utålmodig ved sin server
\scene Konsulenterne kommer ind.
\begin{sketch}
\says{K1} Goddag. Vi kommer direkte fra Oraklet. Er det her
problembarnet holder til?
\says{SO} Ja. Det var søreme godt i kom. Vi har været nede i 4 timer
allerede. Jeg har prrøvet alt hvad jeg kunne finde på. Den vil godt
poste, men ikke pinge.
\says{K2} Aha, javel. Og hvad er det så du har prøvet.

\says{SO} Databasen svarer ikke, men loggen siger alt er i orden. Jeg restartet servicen, jeg har zappet servicen og jeg endda prøvet at reboote!

\says{K1} Rebooote! Det er jo ren overtro at det virker.

\says{K2} Ja, hvis det ikke virkede før, så virker det jo nok heller ikke efter en genstart når du ikke har ændret noget.

\says{SO} Ja, men alligevel. Jeg har altså også læst manualen først.

\says{K1} Manualen?

\says{K2} Det er du jo slet ikke rustet til.

\says{K1} Det var godt du ringede til os. Lad os så komme til.

\says{SO} Ja. Det er herovre...

\says{K2} (afbrryder) Men inden vi kan komme i gang skal du lige underskrive denne her lille Non-Discloosure-Agreement. (Hiver en gigantisk kontrakt op)

\says{K1} Ja, man skulle jo nødig afsløre Oraklets forretningshemmmeligheder.

\act{K2 begynder at finde gummihandkser og stetoskop frem. Han tager
handsker på og lytter på serveren med stetoskopet.}

\says{SO} Det var goodt nok meget. Skal jeg virkelig læse alt det?

\says{K1} Nej. Du skal bare skrive under.

\says{SO} Jeg skriver ikke under på noget jeg ikke har læst.

\says{K1} Ok. Så skal vi bede dig gå udenfor mens vi arbejder.

\says{SO} Men...

\says{K1} Vi kan diskutere om du skal gå undenfor døren for 10.000 kroner i timen, eller vi kan komme i gang med at arbejde. Dit valg.

\act{SO (lister ud)}

\says{K1} (henvendt til K1) Han er gået.

\act{K2 ånder lettet op og hiver handsker af igen.}

\says{K2} Lad os så komme i gang.

\act{K1 og K2 smører ærmerne op en ad gangen og klapper imellem.}

\act{K1 og K2 tager antistatiske armbånd på. På begge arme. De klappe
imellem.}

\act{K1 og K2 slår kraven op.}


\act{K1 og K2 går hen foran serveren med ryggen til publikum.}
De bukker sig begge ned som om de skulle have fat i et panel i bunden.

\scene{Der sker ikke rigtig noget i 2-3 sekunder.}

\act{De begynder ganske langsomt at brumme.}
\act{Mens de laver spiritfingers mod serveren rejser de sig langsomtt op mens
brummen bliver højere og mere skinger.}

\act{Når de har nået vandret med hænderne stikker de i et råb mens
dekaster hænderne i vejret. Råbet ffortsætter i en indianersang.}

\act{De danser hver deres retning rundt om serveren mens de påkalder sig
højere magter.}

\act{Når de er nået 270 grader rundt om serveren vender de rygen til den
og samler en masse energi som de pludseligt og samtidigt vender sig om
og kaster mod serveren.}

\act{Der bliver stille mens de står og afventer og får vejret.}

\act{Da der ikke sker noget går den ene hen og banker en knytnæve på
serveren.}

\scene{Serveren booter.}

\act{De skynder sig at sætte deres tøj pænt igen og SysOp kommer ind.}

\says{SO} Årh. Vildt! Hvad gjorde I?

\says{K1} Man skal bare have en videnskabelig analytisk tilgang til problemet.

\says{K2} Med lidt wiretapping og lowlevel kernekald er årsagen jo
åbenlys.

\says{K1} Og derfra er løsningen jo triviel.

\says{SO} Fedt. Tak skal I have.

\end{sketch}
\scene{Tæppe for, Lys ned.}
