\begin{roles}

  \role{B} Kunde
  \role{U} Mand der ligger på gulvet

\begin{props}
  \prop{Fejlbesked} Et stykke papir
\end{props}

\begin{sketch}

\scene En mand ligger på gulvet (U). En mand (B) står ved siden af ham. En
mand (A) kommer ind.

\says{A} Ja goddag', jeg vil gerne returnere en værdi. Den virker ikke.

\says{B} Desværre Hr., det kan De ikke. Lageret er fyldt op med værdier.

\says{A} Fyldt op?

\says{B} Ja. Lageret er meget værdifuldt.

\says{A} Nå. Jamen så må jeg jo reklamere.

\says{B} Reklamere? For hvad?

\says{A} Her, tag denne her fejlbesked.

\says{B} Fejlbesked?

\says{A} Ja. Jeg kunne ikke returnere min værdi, derfor returnerer jeg denne
fejlbesked. \act{Overrækker fejlbeskeden}

\says{B}[kigger på fejlbeskeden] Hm. Denne fejlbesked kan jeg ikke håndtere. Jeg
rejser lige en undtagelse.

\scene Hjælper U til at rejse sig op.

\says{U} Goddag. De er nu i undtagelsestilstand. Må jeg have lov til at se
billedidentifikation.

\says{A} Du er da ikke nogen Falck-vagt.

\says{B} Hov ikke mere reklame her.

\says{U} Hør her, jeg erklærer undtagelsestilstand!

\says{A} Tage er da ikke særligt ond!

\says{B} Og hvad har det med Elses tilstand at gøre? Er hun syg?

\says{A} Iøvrigt skal vi da også invitere Elses mand If.

\says{B} Ja. Han er meget nIFtig.

\says{U} Så er det godt. Jeg afbryder denne sketch. Den er alt for plat!

\says{A} Jeg troede der var lukket for afbrydelser i undtagelsestilstand.

\says{B} Jeg synes det er synd at afbryde for Else.

\says{A} Nejnej, det hedder "If... Then... Else..." ikke "For... Else..."

\says{B} Forelsket i hvem?

\says{U} STOP! HALT! EXIT!

\says{A,B} T! Og vi hedder ikke Stophaltek!

\scene Tæppe

\end{sketch}
